% v2 model (IWAI26, trust-centric refactor): a parameterised regression
% world, per-agent Gaussian posteriors over a theory parameter theta, and
% one Bayesian learned relational object --- the Gamma-conjugate trust
% matrix Gamma. The patronage flow W and the per-step resource state r
% are DETERMINISTIC READOUTS of Gamma (W = row-normalised Gamma; inflow
% share = aggregate incoming trust). The cost of mind-change lambda is
% retained as the Hyland-style scalar on the socially-induced affective
% utility U whose content is alignment with resource-rich neighbours ---
% it is NOT the same object as the channel precision Gamma.

% ---------- The cost of mind-change (separate from trust) -----------
% lambda scales the socially-induced affective utility U whose content
% is the alignment of the agent's belief with the beliefs of resource-
% rich neighbours. This is the Mal-faithful reading: U is where
% preferences live; preferences are induced by the resource topology,
% not private/abstract.
\newcommand{\stub}{\lambda}                      % cost of mind-change
\newcommand{\mindcost}[1]{\lambda_{#1}}          % per-agent cost of mind-change
\newcommand{\utility}{U}                         % socially-induced affective utility

% ---------- The resource topology (Sec. formal, resources) ----------
% Each agent carries a scalar resource state r_i. Resources flow on the
% graph through a patronage matrix W that is a deterministic readout of
% the trust matrix (W_{j->i} = row-normalised Gamma_{j->i}), and the
% exogenous inflow at rate R_in is distributed by aggregate incoming
% trust (inflowshare_i = sum_j Gamma_{j->i} / total). One Bayesian
% object (Gamma); the resource topology is a readout.
\newcommand{\resource}[1]{r_{#1}}                % agent's resource state
\newcommand{\resourcevec}{\mathbf{r}}            % population resource vector
\newcommand{\flow}[2]{W_{#1#2}}                  % patronage flow rate j->i (= normalised Gamma)
\newcommand{\flowmat}{\mathbf{W}}                % patronage flow matrix (readout of Gamma)
\newcommand{\inflowrate}{R_{\mathrm{in}}}        % total exogenous inflow rate
\newcommand{\inflowshare}[1]{\pi^{\mathrm{in}}_{#1}}  % share of inflow reaching agent i (aggregate trust)
\newcommand{\outflow}{\alpha}                    % per-agent outflow fraction along W
\newcommand{\rdecay}{\delta}                     % resource decay rate
\newcommand{\gatecost}{c}                        % resource-gated experiment cost c(x; r_i)
\newcommand{\fishercost}{\gamma}                 % cost-Fisher coupling steepness (Axis C)
